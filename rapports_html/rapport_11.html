Trains de Paris

SQL> SELECT t.train_id || ' ' || s1.city || ' - ' || s2.city "Trains au départ de Paris", pass_name "Titre d'abonnement", ROUND(t.price-t.price*discount_pct/100, 2) "Tarif en semaine",
2 ROUND(t.price-t.price*discount_we_pct/100, 2) "Tarif en week-end"
3 FROM t_train t
4 CROSS JOIN t_pass p
5 JOIN t_station s1
6 ON t.departure_station_id = s1.station_id
7 JOIN t_station s2
8 ON t.arrival_station_id = s2.station_id
9 WHERE s1.city = 'Paris'
10 ORDER BY train_id, pass_id;
